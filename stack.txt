#include<stdio.h>
#include<stdlib.h>
#include<conio.h>
#define SIZE=0
void push(int);
void pop();
void display();
int stack[SIZE],top=-1;
void main()
{
    int value,choice;
     while(1)
    {
        printf("\n MENU");
        printf("\n 1.PUSH \n 2.DISPLAY \n 3.POP \n 4.EXIT");
        printf("\n Enter your choice:");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:printf("\n Enter the value to be inserted:");
            scanf("%d",&value);
            push(value);
            break;
            case 2:display();
            break;
            case 3:pop();
            break;
            case 4:exit(0);
            default: printf("\n Invalid choice");

        }
    }

}
void push(int value)
{
    if(top==SIZE-1)
         printf("\n Stack is full");
    else
        top++;
    stack[top]=value;
     printf("\n Insertion successfull");
}
void display()
{
    int i;
    if(top==-1)
         printf("\n Stack is Empty");
    else
    {
         printf("\n Stack elements are:");
         for(i=top;i>=0;i--)
             printf("\n%d",stack[i]);

    }
}
void pop()
{
    if(top==-1)
         printf("\n stack is Empty");
    else
    {
         printf("\n Deleted:%d",stack[top]);
         top--;
    }
    }
